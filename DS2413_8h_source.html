43 /*
44  * Provides access to a OneWire-addressable dual-channel I/O device.
45  * The channel latch can be set to on (false) or off (true).
46  * When a channel is off (PIOx=1), the channel state can be sensed. This is the power on-default.
47  *
48  * channelRead/channelWrite reads and writes the channel latch state to turn the output transistor on or off
49  * channelSense senses if the channel is pulled high.
50  */
51 class DS2413
52 {
53 public:
54

57  }

60  * Initializes this ds2413.
61  * /param oneWire The oneWire bus the device is connected to
62  * /param address The oneWire address of the device to use.
63  */
64  void init(OneWire* oneWire, DeviceAddress address)
65  {
66  this->oneWire = oneWire;
67  memcpy(this->address, address, sizeof(DeviceAddress));
68  }

78  /*
79  * Determines if the device is connected. Note that the value returned here is potentially stale immediately on return,
80  * and should only be used for status reporting. In particular, a return value of true does not provide any guarantee
81  * that subsequent operations will succeed.
82  */
83  bool isConnected()
84  {
85  return validAddress(oneWire, this->address) && accessRead()>=0;
86  }

88  // assumes pio is either 0 or 1, which translates to masks 0x1 and 0x2
89  uint8_t pioMask(pio_t pio) { return pio++; }
90 
91  /*
92  * Reads the output state of a given channel, defaulting to a given value on error.
93  * Note that for a read to make sense the channel must be off (value written is 1).
94  */
95  bool channelRead(pio_t pio, bool defaultValue)
96  {
97  byte result = channelReadAll();
98  if (result<0)
99  return defaultValue;
100  return (result & pioMask(pio));
101  }
102 
103 #if DS2413_SUPPORT_SENSE
104  /*
105  * Reads the output state of a given channel, defaulting to a given value on error.
106  * Note that for a read to make sense the channel must be off (value written is 1).
107  */
108  bool channelSense(pio_t pio, bool defaultValue)
109  {
110  byte result = channelSenseAll();
111  if (result<0)
112  return defaultValue;
113  return (result & pioMask(pio));
114  }
115 
116  uint8_t channelSenseAll()
117  {
118  byte result = accessRead();
119  // save bit3 and bit1 (PIO
120  return result<0 ? result : ((result&0x4)>>1 | (result&1));
121  }
122 
123 #endif
124  /*
125  * Performs a simultaneous read of both channels.
126  * /return <0 if there was an error otherwise bit 1 is channel A state, bit 2 is channel B state.
127  */
128  uint8_t channelReadAll()
129  {
130  byte result = accessRead();
131  // save bit3 and bit1 (PIO
132  return result<0 ? result : ((result&0x8)>>2 | (result&2)>>1);
133  }
134 
135  /*
136  * Writes to the latch for a given PIO.
137  * /param set 1 to switch the pin off, 0 to switch on.
138  */
139  bool channelWrite(pio_t pio, bool set)
140  {
141  bool ok = false;
142  byte result = channelReadAll();
143  if (result>=0) {
144  uint8_t mask = pioMask(pio);
145  if (set)
146  result |= mask;
147  else
148  result &= ~mask;
149  ok = channelWriteAll((uint8_t)result);
150  }
151  return ok;
152  }

163  static bool validAddress(OneWire* oneWire, DeviceAddress deviceAddress)
164  {
165  return deviceAddress[0] && (oneWire->crc8(deviceAddress, 7) == deviceAddress[7]);
166  }

169  /*
170  * Fetches the address at the given enumeration index.
171  * Only devices matching the 2413 family ID are considered.
172  * /param deviceAddress the address found
173  * /param index The 0-based device index to return
174  * /return true if the device was found and is the address valid.
175  */
176 #if DS2413_DYNAMIC_ADDRESS
177  static bool getAddress(OneWire* oneWire, uint8_t* deviceAddress, uint8_t index)
178  {
179  oneWire->reset_search();
180 
181  for (uint8_t pos = 0; deviceAddress[0] = 0, oneWire->search(deviceAddress); )
182  {
183  if (deviceAddress[0]==DS2413_FAMILY_ID)
184  {
185  if (pos++ == index)
186  return true;
187  }
188  }
189  return false;
190  }
191 #endif

194  /*
195  * Read all values at once, both current state and sensed values. The read performs data-integrity checks.
196  * Returns a negative result if the device cannot be read successfully within the given number of tries.
197  * The lower 4-bits are the values as described under PIO ACCESSS READ [F5h] in the ds2413 datasheet:
198  * b0: PIOA state
199  * b1: PIOA output latch state
200  * b2: PIOB state
201  * b3: PIOB output latch state
202  */
203  byte accessRead(uint8_t maxTries=3);
204 
205  /*
206  * Writes the state of all PIOs in one operation.
207  * /param b pio data - PIOA is bit 0 (lsb), PIOB is bit 1
208  * /param maxTries the maximum number of attempts before giving up.
209  * /return true on success
210  */
211  bool accessWrite(uint8_t b, uint8_t maxTries=3);
