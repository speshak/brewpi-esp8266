189  void sendSingleItemJsonMessage(const char prefix, const JsonDocument &doc) {
190  if (prefix) {
191  print(prefix);
192  print(':');
193  }
194 
195  // Pull the device object from the document array and promote it to be root
196  DynamicJsonDocument shallowDoc(doc.capacity());
197 
198  for (auto kvp : doc.as<JsonArray>()[0].as<JsonObject>()) {
199  shallowDoc[kvp.key()] = kvp.value();
200  }
201 
202  serializeJson(shallowDoc, stream);
203  printNewLine();
204  }
