126  void step() {
127  if (enabled)
128  {
129 
130  heating = tempControl.stateIsHeating();
131  cooling = tempControl.stateIsCooling();
132  doorOpen = PSensor(tempControl.door)->sense();
133  // with no serial and no calculation here we get 1500-2000x speedup
134  // with this code enabled, around 1300x speedup
135  // with serial, drops to 300x speedup
136  double fermDiff = beerFerment();
137  double heatingDiff = chamberHeating();
138  double coolingDiff = chamberCooling();
139  double doorDiff = doorLosses();
140 
141  double newBeerTemp = beerTemp + fermDiff;
142  double newFridgeTemp = fridgeTemp + heatingDiff + coolingDiff + doorDiff;
143 
144  TempPair beerTx;
145  TempPair roomTx;
146  currentRoomTemp = roomTemp();
147  chamberBeerTransfer(fridgeTemp, beerTemp, beerTx);
148  chamberRoomTransfer(fridgeTemp, currentRoomTemp, roomTx);
149 
150  newFridgeTemp += beerTx.t1 + roomTx.t1; // transfer from beer and environment
151  newBeerTemp += beerTx.t2;
152 
153  fridgeTemp = newFridgeTemp;
154  beerTemp = newBeerTemp;
155 
156  time += 1;
157  }
158  updateSensors();
159  }

251  double roomTemp()
252  {
253  if (minRoomTemp==maxRoomTemp)
254  return minRoomTemp;
255 
256  unsigned long secondsInADay = 60*60*24UL;
257  double p = (double(time%secondsInADay)/double(secondsInADay))*(TWO_PI);
258  double s = sin(p);
259  double mid = (minRoomTemp+maxRoomTemp)/2;
260  double half = mid-minRoomTemp;
261  double r = mid + s*half;
262  return r;
263  }

309  /* Compute the heat transferred between two heat energy potentials. Returns the temperature change
310  of each potential. */
311  void heatTransfer(const HeatPotential& p1, const HeatPotential& p2, double k, TempPair& result)
312  {
313  double e1 = (p2.temp-p1.temp)*k; // the energy transferred - from p2 to p1
314  result.t1 = e1/p1.capacity; // change in temperature for the increase in energy
315  result.t2 = -e1/p2.capacity;
316  }
