95 inline bool isAssignable(DeviceType type, DeviceHardware hardware)
96 {
97  return (hardware==DEVICE_HARDWARE_PIN && (type==DEVICETYPE_SWITCH_ACTUATOR || type==DEVICETYPE_SWITCH_SENSOR))
98 #if BREWPI_DS2413
99  || (hardware==DEVICE_HARDWARE_ONEWIRE_2413 && (type==DEVICETYPE_SWITCH_ACTUATOR || (DS2413_SUPPORT_SENSE && type==DEVICETYPE_SWITCH_SENSOR)))
100 #endif
101  || (hardware==DEVICE_HARDWARE_ONEWIRE_TEMP && type==DEVICETYPE_TEMP_SENSOR)
102  || (hardware==DEVICE_HARDWARE_NONE && type==DEVICETYPE_NONE);
103 }
