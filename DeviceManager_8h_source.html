102 inline bool isAssignable(DeviceType type, DeviceHardware hardware)
103 {
104  return (hardware==DEVICE_HARDWARE_PIN && (type==DEVICETYPE_SWITCH_ACTUATOR || type==DEVICETYPE_SWITCH_SENSOR))
105 #if BREWPI_DS2413
106  || (hardware==DEVICE_HARDWARE_ONEWIRE_2413 && (type==DEVICETYPE_SWITCH_ACTUATOR || (DS2413_SUPPORT_SENSE && type==DEVICETYPE_SWITCH_SENSOR)))
107 #endif
108  || (hardware==DEVICE_HARDWARE_ONEWIRE_TEMP && type==DEVICETYPE_TEMP_SENSOR)
109  || (hardware==DEVICE_HARDWARE_NONE && type==DEVICETYPE_NONE);
110 }

258  int8_t enumerateActuatorPins(uint8_t offset)
259  {
260 #if BREWPI_ACTUATOR_PINS && defined(ARDUINO)
261 #if BREWPI_STATIC_CONFIG<=BREWPI_SHIELD_REV_A

263  case 0: return heatingPin;

265  default:
266  return -1;
267  }
268 #elif BREWPI_STATIC_CONFIG>=BREWPI_SHIELD_REV_C
269  switch (offset) {
270  case 0: return actuatorPin1;
271  case 1: return actuatorPin2;
272  case 2: return actuatorPin3;
273  case 3: return actuatorPin4;
274  default: return -1;
275  }
276 #endif
277 #endif
278  return -1;
279  }
